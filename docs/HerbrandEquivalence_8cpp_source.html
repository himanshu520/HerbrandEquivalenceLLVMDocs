354  void createCFG(Function &F) {
356  // First find the set of reachable basic blocks by
357  // performing BFS from the starting basic block
359 
360  std::set<BasicBlock *> reachableBB;
361  std::queue<BasicBlock *> q;
362 
363  // the list of basic blocks in the same order as
364  // visited by BFS, for same traversal later again
365  std::vector<BasicBlock *> bfsOrder;
366 
367  // mark the starting basic block as reachable
368  // and also push it into the queue for BFS
369  q.push(&F.front()), reachableBB.insert(&F.front());
370 
371  while(not q.empty()) {
372  BasicBlock *bb = q.front();
373  q.pop(), bfsOrder.push_back(bb);
374 
375  for(BasicBlock *nbb : successors(bb)) {
376  if(reachableBB.find(nbb) == reachableBB.end())
377  reachableBB.insert(nbb), q.push(nbb);
378  }
379  }
380 
382  // Now again visit the basic blocks in the same
383  // order assigning an index for nodes in the control
384  // flow graph vector `CFG` corresponding to
385  // instructions and confluence points
387 
388  // some basic blocks might have just one
389  // predecessor, in that case they don't require
390  // a confluence node in the control flow graph.
391  // This set keeps track of all the basic blocks
392  // that require a confluence node
393  std::set<BasicBlock *> confBlocks;
394 
395  // keeps track of current `CFG` index. Index 0 is
396  // assigned to the starting node
397  int curCfgIndex = 1;
398 
399  for(BasicBlock *bb : bfsOrder) {
400  // number of predecessor basic blocks of `bb`,
401  // which are reachable
402  int predSz = 0;
403  for(BasicBlock *nbb : predecessors(bb)) {
404  if(reachableBB.find(nbb) != reachableBB.end())
405  predSz++;
406  }
407 
408  // if this basic block requires a confluence node,
409  // first assign an index to it
410  if(predSz > 1)
411  confBlocks.insert(bb), curCfgIndex++;
412 
413  // now assign nodes for the instructions in the
414  // basic block
415  for(Instruction &I : (*bb))
416  CfgIndex[&I] = curCfgIndex++;
417  }
418 
420  // Now actually create control flow graph nodes by
421  // filling `CFG` vector. Note that the consistency
422  // between `CfgIndex` and actual index of node for
423  // instructions in `CFG` vector is implicitly
424  // maintained
426 
427  // START node
428  CFG.push_back({CfgNodeTy::START, nullptr, std::vector<int>()});
429 
430  // `CFG` nodes that are predecessors of the special END
431  // END instruction. Last instruction of the basic blocks
432  // that have no successors belong to this category. Only
433  // if this vector is non-empty, an END node is created
434  std::vector<int> predsEnd;
435 
436  for(auto bb : bfsOrder) {
437  // holds index of predecessor nodes for instructions
438  // in this block
439  int predIndex;
440 
441  // if the block requires a confluence point,
442  // first push it into the `CFG` vector
443  if(confBlocks.find(bb) != confBlocks.end()) {
444  std::vector<int> preds;
445  for(BasicBlock *nbb : predecessors(bb)) {
446  if(reachableBB.find(nbb) != reachableBB.end())
447  preds.push_back(CfgIndex[&nbb->back()]);
448  }
449  CFG.push_back({CfgNodeTy::CONFLUENCE, nullptr, preds});
450 
451  // initialise `predIndex`
452  predIndex = CFG.size() - 1;
453  } else {
454  // here only `predIndex` is
455 
456  // this initialisation is for the first basic block
457  // for which the `if` condition inside `for` never
458  // becomes true
459  predIndex = 0;
460 
461  for(BasicBlock *nbb : predecessors(bb))
462  if(reachableBB.find(nbb) != reachableBB.end()) {
463  // if `break` is removed, even then the `if`
464  // block should be entered only once (except
465  // for the first basic block for which it is
466  // entered never)
467  predIndex = CfgIndex[&nbb->back()];
468  break;
469  }
470  }
471 
472  // now insert nodes corresponding to the instructions
473  for(Instruction &I : (*bb)) {
474  CFG.push_back({CfgNodeTy::TRANSFER, &I,
475  std::vector<int>({predIndex})});
476 
477  // update `predIndex`
478  predIndex = CFG.size() - 1;
479 
480  // this assertion must be passed for consistency
481  // of this function
482  assert(predIndex == CfgIndex[&I]);
483  }
484 
485  if(succ_empty(bb)) predsEnd.push_back(CfgIndex[&bb->back()]);
486  }
487 
488  // now create node corresponding to END, if required
489  if(not predsEnd.empty()) {
490  CFG.push_back({CfgNodeTy::END, nullptr, predsEnd});
491  }
492  }

660  // map to store equivalent indexes (have same value)
661  // in the first partition

664  mp[first[i]].push_back(i);
665

667  // partition are also equivalent in the second.

704  int findSet(std::vector<int> const &partition, ExpressionTy const &exp) {
705  // operator and set identifiers corresponding to left and right
706  // subexpressions at the current program point
707  char op = std::get<0>(exp);
708  int leftSetId = partition[IndexExp[EXP(std::get<1>(exp))]];
709  int rightSetId = partition[IndexExp[EXP(std::get<2>(exp))]];
710 
711  // checking if a set representing the expression
712  // already exists
713  std::tuple<char, int, int> tup = {op, leftSetId, rightSetId};
714  auto it = Parent.find(tup);
715 
716  // if the set already exists return its identifier,
717  // otherwise return new set identifier and update
718  // `Parent` map with this information
719  if(it == Parent.end()) return Parent[tup] = SetCnt++;
720 
721  return it->second;
722  }

795  void transferFunction(int cfgIndex) {
796  // current partition vector
797  std::vector<int> &partition = Partitions[cfgIndex];
798 
799  // first copy predecessor partition into current partition
800  partition = Partitions[CFG[cfgIndex].predecessors[0]];
801 
802  // if the current partition has any index with value -1, it
803  // means that it represents the TOP element and it has to be
804  // left as such without any modifications
805  if(partition[0] == -1) return;
806 
807  // if the node is the END CFG node, return
808  if(CFG[cfgIndex].NodeTy == CfgNodeTy::END) return;
809 
810  // `changedExp` is expression (ie. a variable) which
811  // has been assigned value and `changedToExp` is the
812  // expression which has been assigned to it. If
813  // operator in `changedToExp` is `#` - it symbolises
814  // non-deterministic assignment
815  Instruction *inst = CFG[cfgIndex].instPtr;
816  ExpressionTy changedExp, changedToExp;
817 
818  if(isa<LoadInst>(inst)) {
819  changedExp = EXP(inst);
820  changedToExp = EXP(inst->getOperand(0));
821  } else if(isa<StoreInst>(inst)) {
822  changedExp = EXP(inst->getOperand(1));
823  changedToExp = EXP(inst->getOperand(0));
824  } else if(isa<BinaryOperator>(inst)) {
825  char op = getOpSymbol(inst->getOpcodeName());
826  Value *leftOp = inst->getOperand(0);
827  Value *rightOp = inst->getOperand(1);
828  changedExp = EXP(inst);
829  changedToExp = {op, leftOp, rightOp};
830  } else if(isa<CallInst>(inst)) {
831  changedExp = EXP(inst);
832  // here `#` symbolises non-deterministic assignment
833  changedToExp = {'#', nullptr, nullptr};
834  } else {
835  // in this case the current instruction has not
836  // modified the partition, so the function returns
837  return;
838  }
839 
840  if(std::get<0>(changedToExp) == '#') {
841  // if it is a non-deterministic assignment,
842  // then create a new set identifier
843  partition[IndexExp[changedExp]] = SetCnt++;
844  } else {
845  // assign the `changedExp`, the set identifier of
846  // `changedToExp`
847  partition[IndexExp[changedExp]] = partition[IndexExp[changedToExp]];
848  }
849 
850  // update set identifiers for two length expressions
851  // involving `changedExp`. Here set identifiers of all
852  // two length expressions are assigned values, but only
853  // those involving `changedExp` are updated
854  for(auto &el : IndexExp) {
855  if(std::get<0>(el.first) == '\0') continue;
856  partition[el.second] = findSet(partition, el.first);
857  }
858  }

869  void confluenceFunction(int cfgIndex) {
870  // vector of predecessor CFG node indexes
871  std::vector<int> &predecessors = CFG[cfgIndex].predecessors;
872 
873  // if all the predecessors partition represents TOP
874  // element then their confluence is also TOP element.
875  // So, the current partition need not be modified as
876  // by initialisation the current partition should also
877  // hold TOP partition
878  bool cont = false;
879  for(int pred : predecessors)
880  cont |= (Partitions[pred][0] != -1);
881  if(not cont) return;
882 
883  // the current partition
884  std::vector<int> &partition = Partitions[cfgIndex];
885 
886  // to check which expressions has already been processed
887  std::vector<bool> accessFlag(IndexExp.size(), false);
888 
889  // process all the expressions one by one
890  for(auto &el : IndexExp) {
891  int elIdx = el.second;
892 
893  // continue if the expression is already processed
894  if(accessFlag[elIdx]) continue;
895 
896  // mark it as processed
897  accessFlag[elIdx] = true;
898 
899  // if the set identifier for the expression in all
900  // the predecessors is same it is assigned the same
901  // set identifier; else a new set identifier is
902  // created and assigned to all the expressions which
903  // belongs to the same equivalence class as the
904  // current expression across all the predecessors and
905  // they are also marked processed
906  int elSetId = -1;
907  bool flag = true;
908  for(int pred : predecessors) {
909  int elSetIdPred = Partitions[pred][elIdx];
910  if(elSetIdPred == -1) continue;
911  else if(elSetId == -1) elSetId = elSetIdPred;
912  else if(elSetId != elSetIdPred) flag = false;
913  }
914 
915  if(flag) partition[elIdx] = elSetId;
916  else {
917  std::set<int> intersection, elClassPred;
918  for(int i = 0; i < (int)IndexExp.size(); i++)
919  intersection.insert(i);
920 
921  for(int pred : predecessors) {
922  getClass(Partitions[pred], elIdx, elClassPred);
923  setIntersect<int>(intersection, elClassPred);
924  }
925 
926  int newSetId = SetCnt++;
927  for(int nel : intersection)
928  accessFlag[nel] = true, partition[nel] = newSetId;
929  }
930  }
931 
932  // now update `Parent` map
933  for(auto &el : IndexExp) {
934  char op = std::get<0>(el.first);
935  if(op == '\0') continue;
936 
937  int leftSetID = partition[IndexExp[EXP(std::get<1>(el.first))]];
938  int rightSetID = partition[IndexExp[EXP(std::get<2>(el.first))]];
939 
940  auto tup = std::make_tuple(op, leftSetID, rightSetID);
941  // this assertion must be passed for the consistency of the algorithm
942  assert(Parent.find(tup) == Parent.end() or
943  Parent[tup] == partition[el.second]);
944 
945  Parent[tup] = partition[el.second];
946  }
947  }

956  void HerbrandAnalysis(Function &F) {
957  PRINT_HEADER("Herbrand Equivalence Computation");
958  errs() << "\n";
959 
960  // assign index to expressions arbitrarily
961  assignIndex(F);
962 
963  // initialise partition vector with -1 for each program
964  // points and each expression - this stands for TOP
965  // partition at each program point. Note that any element
966  // of partition vector being -1 means that whole vector
967  // holds -1 and represents TOP partition
968  Partitions.assign(CFG.size(),
969  std::vector<int>(IndexExp.size(), -1));
970 
971  // initialise starting partition for START node
972  findInitialPartition(Partitions[0]);
973 
974  PRINT_HEADER("Initial Partition");
975  printPartition(Partitions[0]);
976  errs() << "\n\n\n";
977 
978  bool converged = false;
979  int iterationCtr = 0;
980 
981  // repeat while convergence
982  while(not converged) {
983  PRINT_HEADER("Iteration " + std::to_string(++iterationCtr));
984  converged = true;
985 
986  // for all program points (nodes in CFG) except
987  // START apply transfer/confluence function as
988  // applicable
989  for(int i = 1; i < (int)CFG.size(); i++) {
990  std::vector<int> oldPartition = Partitions[i];
991  std::vector<int> &predecessors = CFG[i].predecessors;
992  Instruction *inst = CFG[i].instPtr;
993 
994  errs() << '[' << i << "] : ";
995 
996  bool isConfluence = (CFG[i].NodeTy == CfgNodeTy::CONFLUENCE);
997  if(CFG[i].NodeTy == CfgNodeTy::END and predecessors.size() > 1)
998  isConfluence = true;
999 
1000  if(isConfluence) {
1001  // if CFG node corresponds to a confluence point
1002  if(CFG[i].NodeTy == CfgNodeTy::CONFLUENCE)
1003  errs() << "Confluence Point => ";
1004  else errs() << "END => ";
1005 
1006  errs() << "[Predecessors :";
1007  for(auto el : predecessors) {
1008  errs() << ' ' << el << '('
1009  << CFG[el].instPtr->getParent()->getName()
1010  << ')';
1011  }
1012  errs() << "]\n\t";
1013 
1014  confluenceFunction(i);
1015  } else {
1016  // if CFG node corresponds to a transfer point
1017 
1018  if(CFG[i].NodeTy == CfgNodeTy::TRANSFER) {
1019  errs() << "Transfer Point => ["
1020  << inst->getParent()->getName() << "] "
1021  << (*inst) << ' ';
1022  } else errs() << "END => ";
1023 
1024  errs() << "[Predecessors : " << predecessors[0] << "]\n\t";
1025 
1026  transferFunction(i);
1027  }
1028 
1029  printPartition(Partitions[i]);
1030  errs() << "\n\n";
1031 
1032  // update convergence flag
1033  if(not samePartition(oldPartition, Partitions[i]))
1034  converged = false;
1035  }
1036  errs() << "\n\n";
1037  }
1038  }
