576  vector<IDstruct *> tempPartition(numExps, nullptr);
577  vector<Instruction *> predInsts;
578  findPredecessors(I, predInsts);
579 
580  vector<bool> accessFlag(CuV.size(), false);

584  bool flag = true;
585 
586  for(int i = 1; i < (int)predInsts.size() && flag; i++)
587  if(Partitions[predInsts[0]][el.second] != Partitions[predInsts[i]][el.second])
588  flag = false;
589 
590  if(flag) tempPartition[el.second] = Partitions[predInsts[0]][el.second];
591  else {
592  set<Value *> vV, intersection(CuV);

684  bool runOnFunction(Function &F) override {
685  assignIndex(F);
686  vector<IDstruct *> initialPartition;
687  findInitialPartition(initialPartition);
688  errs() << string(150, '=') << "\nInitial Partition " << "\n" << string(150, '=') << "\n";
689  printPartition(initialPartition);
690  errs() << "\n\n";
691 
692  for(Instruction &I : instructions(F)) findInitialPartition(Partitions[&I]);
693 
694  bool converged = false;
695  int iterationCtr = 0;
696  while(not converged) {
697  errs() << string(75, '=') << "\nIteration " << ++iterationCtr << "\n" << string(75, '=') << "\n";
698  converged = true;
699 
700  Instruction *prevInst = nullptr;
701  for(Instruction &I : instructions(F)) {
702  errs() << I << "\n";
703  // if(Partitions.find(&I) == Partitions.end()) Partitions[&I] = vector<IDstruct *>(numExps, nullptr);
704 
705  vector<IDstruct *> oldPartition = Partitions[&I];
706  if(prevInst == nullptr)
707  AssignStatement(Partitions[&I], initialPartition, I);
708  else if(&I.getParent()->front() == &I)
709  confluence(Partitions[&I], I);
710  else AssignStatement(Partitions[&I], Partitions[prevInst], I);
711 
712  prevInst = &I;
713  if(not samePartition(oldPartition, Partitions[&I])) converged = false;
714  printPartition(Partitions[&I]);
715  errs() << "\n\n\n";
716  }
717  }
718  return false;
719  }
